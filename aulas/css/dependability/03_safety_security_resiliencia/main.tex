\section{Motivação e mapa conceitual}

\begin{frame}{Motivação}
\begin{itemize}
  \item Sistemas modernos são \textbf{ciberfísicos} e \textbf{conectados}: falhas + ataques podem gerar \textbf{danos físicos}.
  \item Dependability (confiabilidade) não é ``só bug'': envolve \textbf{hardware, software, rede, humano, processo}.
  \item \textbf{Convergência}: safety (acidental) + security (malicioso) + resiliência (sobrevivência e recuperação).
  \item A indústria já formalizou isso: a \textbf{PSA} (\textit{Platform Security Architecture}) da Arm define um framework de análise de ameaças, requisitos de segurança e implementação de referência para dispositivos conectados --- mostrando que \textbf{security by design} é exigência de mercado, não opcional.
\end{itemize}
\vspace{0.4em}
\small
\textbf{Objetivo da aula:} fornecer um \textbf{framework} para identificar e justificar requisitos (safety/security) e decisões arquiteturais (resiliência).

\begin{frame}{Dependability/Trustworthiness: visão unificadora (taxonomia clássica)}
\begin{block}{Ideia-chave}
\textbf{Dependability} é um conceito guarda-chuva que inclui atributos como:
\textit{reliability, availability, safety, integrity, maintainability} (e, ao considerar security, entra \textit{confidentiality}).
\end{block}

\begin{itemize}
  \item \textbf{Atributos (o que queremos)}: disponibilidade, confiabilidade, segurança funcional, integridade, etc.
  \item \textbf{Ameaças (o que pode dar errado)}: \textit{fault} $\rightarrow$ \textit{error} $\rightarrow$ \textit{failure}; e ataques.
  \item \textbf{Meios (como alcançamos)}: prevenção, tolerância a falhas, remoção de falhas, previsão/estimativa.
\end{itemize}

\small
\hfill \textit{Fonte: Avizienis et al., IEEE TDSC 2004.}
%% [AULA] Reforce: “taxonomia” serve para organizar o pensamento e evitar checklist aleatório.
\end{frame}

\begin{frame}{Safety vs Security: definição e diferença operacional}
\vspace{-1.5em}
  \begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{tcolorbox}[title=Safety (acidental),colback=white]
\begin{itemize}
  \item Foco: \textbf{evitar dano} a pessoas/ambiente.
  \item Causa típica: \textbf{falha não intencional} (bug, desgaste, erro humano).
  \item Pergunta: ``\textit{o que pode causar um acidente?}'' (hazard analysis)
\end{itemize}
\end{tcolorbox}

\column{0.48\textwidth}
\begin{tcolorbox}[title=Security (malicioso),colback=white]
\begin{itemize}
  \item Foco: \textbf{proteger ativos} contra adversários.
  \item Causa típica: \textbf{ação intencional} (exploit, engenharia social, supply chain).
  \item Pergunta: ``\textit{como um atacante consegue isso?}'' (threat modeling)
\end{itemize}
\end{tcolorbox}
\end{columns}

\vspace{0.3em}
\small
\textbf{Interseção crítica:} ataques podem gerar \textbf{hazards} --- por exemplo, um atacante que compromete a rede CAN de um veículo pode injetar mensagens falsas de frenagem, transformando uma vulnerabilidade de \textit{security} em risco direto à vida (\textit{safety}).
%% [AULA] Discuta “assets” em sistemas ciberfísicos: não só dados; inclui controle, tempo, continuidade, integridade de comando.
\vspace{0.3em}
\scriptsize
\textit{Veja na prática:} \href{https://www.youtube.com/watch?v=MK0SrxBC1xs}{Hacking Jeep}
\end{frame}

\begin{frame}{Ameaças: falhas vs ataques (vocabulário)}
\begin{columns}[T,onlytextwidth]
\column{0.5\textwidth}
\textbf{Mundo de falhas (dependability clássica)}
\begin{itemize}
  \item \textbf{Fault}: defeito/causa (HW/SW/humano).
  \item \textbf{Error}: estado incorreto interno.
  \item \textbf{Failure}: desvio observado do serviço esperado.
\end{itemize}

\column{0.5\textwidth}
\textbf{Mundo de ataques (security)}
\begin{itemize}
  \item \textbf{Threat}: adversário + capacidade + intenção.
  \item \textbf{Vulnerability}: fraqueza explorável.
  \item \textbf{Exploit}: método para violar uma propriedade.
\end{itemize}
\end{columns}

\vspace{0.4em}
\begin{block}{Ponte entre os mundos}
\textbf{Ataque} explora \textbf{vulnerabilidade} e induz \textbf{erro} (estado incorreto), resultando em \textbf{falha} (failure) que pode virar \textbf{acidente} (safety).
\end{block}
%% [AULA] Explique por que “bug” e “vuln” não são sinônimos: bug só vira vuln sob modelo de ameaça + superfície de ataque.
\end{frame}

\section{Métricas e trade-offs de dependability}

\begin{frame}{Métricas clássicas: Reliability e Availability}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\textbf{Reliability} (sobrevivência até $t$)
\[
R(t)=P(T>t)
\]
Para taxa de falhas constante $\lambda$ (modelo exponencial):
\[
R(t)=e^{-\lambda t}
\]

\column{0.48\textwidth}
\textbf{Availability} (tempo operacional)
\[
A \approx \frac{MTBF}{MTBF+MTTR}
\]
\begin{itemize}
  \item MTBF: tempo médio entre falhas
  \item MTTR: tempo médio de reparo/recuperação
\end{itemize}
\end{columns}

%% [AULA] Reforce: disponibilidade pode ser alta com falhas frequentes se o reparo for rápido (MTTR baixo).
\end{frame}

\begin{frame}{Integridade, Safety e Security: métricas e “o que medir”}
\begin{itemize}
  \item \textbf{Integrity}: correção e não corrupção (dados/estado/comando).
  \item \textbf{Safety}: probabilidade de \textbf{dano} + severidade (frequentemente via classes de risco).
  \item \textbf{Security}: não há uma única métrica; use \textbf{objetivos} (CIA/AAA), risco, cobertura de controles, e métricas operacionais (MTTD/MTTR).
  \begin{itemize}
    \item \textbf{CIA}: \textit{Confidentiality} (acesso só a quem autorizado), \textit{Integrity} (dados não adulterados), \textit{Availability} (sistema acessível quando necessário).
    \item \textbf{AAA}: \textit{Authentication} (verificar identidade), \textit{Authorization} (controlar permissões), \textit{Accountability} (rastrear ações).
  \end{itemize}
\end{itemize}

\vspace{0.3em}
\begin{block}{Atenção}
Métricas precisam ser \textbf{amarradas a um objetivo} e a um \textbf{modelo de ameaça/uso}. Sem isso viram ``números bonitos''.
\end{block}
%% [AULA] Discuta armadilhas: “contar CVEs” vs risco real; “coverage de teste” vs evidência de propriedade.
\end{frame}

\begin{frame}{Trade-offs típicos}
\begin{itemize}
  \item \textbf{Autenticação} pode aumentar \textbf{latência} (impacto em controle em tempo real).
  \item \textbf{Criptografia} aumenta custo/energia; pode afetar disponibilidade em HW fraco.
  \item \textbf{Redundância} melhora \textbf{availability}, mas pode piorar \textbf{reliability} se introduzir mais componentes/falhas.
\end{itemize}

\begin{block}{Engenharia = justificar trade-offs}
Não é “ter tudo”, é \textbf{definir o que é aceitável} e \textbf{provar com evidência}.
\end{block}
%% [AULA] Puxe o gancho para V&V: evidência vem de testes, análises, inspeções, revisões, medições e auditorias.
\end{frame}

\section{Safety engineering: de hazard a requisito e evidência}

\begin{frame}{Safety: conceitos operacionais}
\begin{itemize}
  \item \textbf{Hazard}: condição/estado do sistema que, combinada com o ambiente, pode levar a dano.
  \item \textbf{Accident/Loss}: dano efetivo (evento de perda).
  \item \textbf{Safety constraint}: restrição que deve ser mantida para evitar hazards.
\end{itemize}

\begin{block}{Safety não é “zero risco”}
É \textbf{reduzir risco a um nível aceitável} (e justificar isso).
\end{block}

%% [AULA] Diferencie “hazard” de “failure”: falha é desvio do serviço; hazard é estado perigoso (pode ocorrer sem falha “visível”).
\end{frame}

\begin{frame}{Ciclo de vida e padrões (o que a indústria exige)}
\begin{itemize}
  \item Safety-critical usa \textbf{ciclo de vida} + \textbf{gestão de configuração} + \textbf{rastreabilidade}.
  \item Padrões variam por domínio: industrial (\textit{IEC 61508}), automotivo (\textit{ISO 26262}), aeronáutico (\textit{DO-178C/ED-12C}).
\end{itemize}

\begin{block}{Artefatos típicos}
Hazard log, safety requirements, arquitetura de mitigação, V\&V plan, evidências, \textbf{safety case}.
\end{block}

%% [AULA] Explique “safety case” como argumento estruturado + evidências (puxar para GSN, sem entrar demais).
%% TODO: inserir figura simples de V-model / safety lifecycle (preferir figura própria via TikZ).
\end{frame}

\begin{frame}{Ferramentas de Hazard Analysis}

\textbf{FMEA} — \textit{Failure Mode and Effects Analysis}
\begin{itemize}
  \item Abordagem \textbf{bottom-up}: parte dos componentes e pergunta ``o que acontece se isso falhar?''
  \item Identifica modos de falha individuais e seus efeitos no sistema.
  \item Adequado para HW/SW com fronteiras bem definidas entre componentes.
\end{itemize}

\vspace{0.6em}

\textbf{FTA} — \textit{Fault Tree Analysis}
\begin{itemize}
  \item Abordagem \textbf{top-down}: parte de um evento indesejado e busca as causas raízes.
  \item Representa combinações lógicas de falhas (portas AND/OR) que levam ao acidente.
  \item Útil para justificar redundâncias e demonstrar que combinações críticas são improváveis.
\end{itemize}

%% [AULA] Mostre como FMEA e FTA se complementam: FMEA descobre modos de falha, FTA mostra como se combinam.
\end{frame}

\begin{frame}{Exemplo de FMEA}
\footnotesize
\begin{tabular}{p{2.0cm} p{2.8cm} p{2.8cm} p{2.8cm}}
\toprule
\textbf{Item} & \textbf{Modo de falha} & \textbf{Efeito} & \textbf{Mitigação}\\
\midrule

Válvula de resfriamento & falha em posição fechada & temperatura sobe sem controle & válvula normalmente aberta (\textit{fail-safe})\\

\bottomrule
\end{tabular}

Mitigação vira \textbf{requisito verificável} (e entra no plano de V\&V).
\end{block}

\end{frame}

\section{Security engineering: de threat model a controles e verificação}

\begin{frame}{Security: propriedades e fronteiras de confiança}
\begin{itemize}
  \item Objetivos clássicos: \textbf{CIA} (Confidentiality, Integrity, Availability) + \textbf{AAA} (Authentication, Authorization, Accounting).
  \item Em embarcados/ciberfísicos, \textbf{integridade de comando} e \textbf{tempo} (freshness) são críticos.
  \item Segurança começa com \textbf{trust boundaries}: onde dados/comandos cruzam domínios.
\end{itemize}

%% TODO: inserir figura (pode ser feita em TikZ) com trust boundaries.
\end{frame}

\begin{frame}{STRIDE: categorias de ameaças}
\footnotesize
\begin{tabular}{p{2.4cm} p{4.8cm} p{3.0cm}}
\toprule
\textbf{Categoria} & \textbf{Descrição} & \textbf{Propriedade violada}\\
\midrule
\textbf{S}poofing & falsificar identidade de usuário ou componente & Authentication\\
\textbf{T}ampering & modificar dados ou código sem autorização & Integrity\\
\textbf{R}epudiation & negar ter realizado uma ação & Accountability\\
\textbf{I}nformation disclosure & expor dados a quem não tem permissão & Confidentiality\\
\textbf{D}enial of Service & tornar o sistema indisponível & Availability\\
\textbf{E}levation of privilege & obter permissões além do autorizado & Authorization\\
\bottomrule
\end{tabular}

\vspace{0.4em}
\small
\textit{STRIDE é um ponto de partida para identificar ameaças — cada categoria orienta perguntas sobre o sistema.}
\end{frame}

\begin{frame}{Threat Modeling}

\begin{enumerate}
  \item \textbf{Modelar o sistema} — desenhar o diagrama de fluxo de dados (DFD) com componentes, fluxos e \textit{fronteiras de confiança} (ex: onde dados cruzam de rede externa para sistema interno).

  \vspace{0.3em}
  \item \textbf{Identificar ameaças} — percorrer cada componente e fluxo com STRIDE: ``o que um atacante pode fazer aqui?''

  \vspace{0.3em}
  \item \textbf{Avaliar e priorizar} — estimar impacto e probabilidade de cada ameaça; focar esforço onde o risco é maior.

  \vspace{0.3em}
  \item \textbf{Definir mitigações} — para cada ameaça priorizada, mapear um controle concreto (ex: autenticação mútua, criptografia, validação de entrada).
\end{enumerate}

\vspace{0.3em}
\begin{block}{Importante}
Threat modeling é \textbf{iterativo}: deve ser revisado a cada mudança relevante de arquitetura.
\end{block}
%% [AULA] Faça um mini-exemplo no quadro: sistema de controle industrial, fronteira OT/IT, ameaça de spoofing no canal de comando.
\end{frame}

\begin{frame}{Risk Assessment em security: processo (NIST SP 800-30)}
\begin{itemize}
  \item Defina: escopo, ativos, missão, tolerância a risco (\textit{risk framing}).
  \item Identifique: \textbf{threat sources/events}, \textbf{vulnerabilities}, \textbf{predisposing conditions}.
  \item Estime: \textbf{likelihood} e \textbf{impact}.
  \item Determine: \textbf{risk} e priorize respostas.
\end{itemize}

\begin{block}{Entrega típica}
Registro de riscos + plano de tratamento (mitigar, transferir, aceitar, evitar) + monitoramento.
\end{block}

\small
\textit{Referência sugerida: NIST SP 800-30 Rev.1.}
%% [AULA] Discuta diferença entre “likelihood de exploração” e “likelihood de ocorrência de ameaça” no contexto do sistema.
\end{frame}


\begin{frame}{Mecanismos típicos (embedded/IoT)}
\begin{itemize}
  \item \textbf{Secure boot} + cadeia de confiança (root of trust).
  \item \textbf{Atualização segura} (assinatura, anti-rollback, A/B, recovery).
  \item \textbf{Proteção de chaves} (HSM/secure element/TPM quando aplicável).
  \item \textbf{Hardening}: superfície mínima, privilégios mínimos, isolamento (MPU/MMU).
  \item \textbf{Observabilidade}: logs auditáveis, telemetria, detecção de anomalia.
\end{itemize}

%% [AULA] Explique por que “criptografia” não substitui arquitetura: ainda precisa isolamento, controle de acesso, boot seguro, etc.
\end{frame}

\begin{frame}{Verificação em security: do requisito ao teste}

\textbf{Exemplo de requisito:} \textit{“toda atualização deve ser autenticada e íntegra.”}

\vspace{0.5em}
Como gerar evidência para esse requisito:

\vspace{0.3em}
\begin{itemize}
  \item \textbf{Revisão de design + threat model} — verificar se a arquitetura cobre a ameaça de atualização maliciosa.

  \vspace{0.2em}
  \item \textbf{Testes dinâmicos} — fuzzing do parser de pacotes, testes de downgrade e rollback forçado.

  \vspace{0.2em}
  \item \textbf{Análise estática e de composição}
  \begin{itemize}
    \item \textbf{SAST}: busca vulnerabilidades no código-fonte sem executar o programa.
    \item \textbf{DAST}: testa o sistema em execução simulando ataques reais.
    \item \textbf{SBOM} + revisão de dependências: lista todos os componentes e verifica vulnerabilidades conhecidas (CVEs).
  \end{itemize}

  \vspace{0.2em}
  \item \textbf{Pentest} orientado ao modelo de ameaça — validação por tentativa de exploração real.
\end{itemize}

%% [AULA] Gancho forte para a próxima aula: V&V, níveis de teste e planejamento para gerar evidência.
\end{frame}

\begin{frame}{Quando security vira safety (e vice-versa)}
\small
\textbf{Security $\Rightarrow$ Safety:}
\begin{itemize}
  \item Em 2015, pesquisadores assumiram controle remoto de um Jeep Cherokee via vulnerabilidade no sistema de infotainment --- acelerando, freando e desligando o motor em rodovia (recall de 1,4 milhão de veículos).
  \item Em 2021, o ataque ransomware à Colonial Pipeline interrompeu o fornecimento de combustível no leste dos EUA, afetando hospitais e serviços de emergência.
\end{itemize}

\vspace{0.3em}
\textbf{Safety $\Rightarrow$ Security:}
\begin{itemize}
  \item Portas de manutenção e modos de bypass criados para safety (ex.: acesso de emergência sem autenticação) tornam-se vetores de ataque se não forem protegidos.
  \item Um gateway automotivo que exige autenticação forte pode \textbf{bloquear comandos de frenagem de emergência} se as credenciais falharem --- a mitigação de security cria um hazard de safety.
\end{itemize}

\vspace{0.3em}
\textbf{Mensagem:} projetar separadamente é arriscado --- o correto é \textbf{co-analisar} interações e trade-offs.
\end{frame}

\section{Resiliência (incl. cyber resiliency) como requisito arquitetural}

\begin{frame}{Resiliência: definição operacional}
\begin{block}{Definição (cyber resiliency)}
Capacidade de \textbf{antecipar}, \textbf{resistir}, \textbf{recuperar} e \textbf{adaptar} a condições adversas, estresses, ataques ou compromissos envolvendo recursos cibernéticos.
\end{block}

\begin{itemize}
  \item Resiliência não substitui safety/security: ela \textbf{complementa} quando a prevenção falha.
  \item Entra como requisitos de: \textbf{degradação graciosa}, \textbf{fail-operational}, \textbf{reconfiguração}, \textbf{continuidade}.
\end{itemize}

\small
\textit{Referência sugerida: NIST SP 800-160 v2r1 (Developing Cyber-Resilient Systems).}
%% TODO: link: https://nvlpubs.nist.gov/nistpubs/SpecialPublications/NIST.SP.800-160v2r1.pdf
\end{frame}

\begin{frame}{Padrões arquiteturais de resiliência}
\begin{columns}[T,onlytextwidth]
\column{0.5\textwidth}
\textbf{Antecipar/Resistir}
\begin{itemize}
  \item segmentação/zonas
  \item least privilege / isolamento
  \item diversidade (N-version, heterogeneidade)
  \item rate limiting, circuit breakers
  \item validação robusta (parsers)
\end{itemize}

\column{0.5\textwidth}
\textbf{Recuperar/Adaptar}
\begin{itemize}
  \item rollback (A/B), recovery mode
  \item checkpoints, reinicialização controlada
  \item degradação graciosa (safe state)
  \item reconfiguração/roteamento alternativo
  \item observabilidade + aprendizado pós-incidente
\end{itemize}
\end{columns}

%% [AULA] Discuta “diversidade” como defesa contra falhas correlacionadas e ataques replicáveis.
\end{frame}

\begin{frame}{Resiliência como “engenharia ao longo do ciclo de vida”}
\begin{itemize}
  \item Resiliência exige processos: governança, gestão de riscos, resposta a incidentes, melhorias contínuas.

\end{itemize}

\section{Modelagem aplicada: PSA (Platform Security Architecture)}

\begin{frame}{PSA: o que é e por que importa}
\begin{itemize}
  \item \textbf{PSA} (\textit{Platform Security Architecture}) é um framework criado pela Arm (2017) para guiar a construção de dispositivos IoT seguros.
  \item Não é só especificação de hardware: cobre \textbf{análise de ameaças}, \textbf{arquitetura}, \textbf{implementação} e \textbf{certificação}.
  \item Agnóstico de implementação --- pode ser aplicado a qualquer chip, software ou dispositivo.
\end{itemize}

\vspace{0.3em}
\begin{block}{As 4 etapas do PSA}
\textbf{Analyze} (modelar ameaças e requisitos) $\rightarrow$
\textbf{Architect} (projetar a segurança) $\rightarrow$
\textbf{Implement} (construir com referência aberta) $\rightarrow$
\textbf{Certify} (validar contra os 10 security goals).
\end{block}
\end{frame}

\begin{frame}{PSA: os 10 Security Goals}
\small
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{enumerate}
  \item \textbf{Unique Device Identity} --- cada dispositivo com identidade única e atestável.
  \item \textbf{Security Lifecycle} --- estados de segurança rastreáveis (fabricação, operação, fim de vida).
  \item \textbf{Attestation} --- evidência verificável das propriedades do dispositivo.
  \item \textbf{Secure Boot} --- apenas código autorizado executa no boot.
  \item \textbf{Secure Update} --- atualizações autênticas e íntegras via OTA (\textit{Over-The-Air}).
\end{enumerate}

\column{0.48\textwidth}
\begin{enumerate}\setcounter{enumi}{5}
  \item \textbf{Anti-Rollback} --- impedir reinstalação de versões antigas vulneráveis.
  \item \textbf{Isolation} --- serviços confiáveis isolados entre si e do código não confiável.
  \item \textbf{Interaction \& Communication} --- comunicação autenticada e cifrada.
  \item \textbf{Secure Storage} --- proteção de chaves, credenciais e dados sensíveis em repouso.
  \item \textbf{Crypto Services} --- primitivas criptográficas fornecidas pelo RoT (\textit{Root of Trust}).
\end{enumerate}
\end{columns}
\end{frame}

\begin{frame}{PSA: RoT e ambientes de processamento}
\begin{itemize}
  \item \textbf{RoT} (\textit{Root of Trust}): componente de hardware/software que ancora as funções de segurança --- identidade, criptografia, attestation e secure boot.
  \item O \textbf{PRoT} (\textit{Platform Root of Trust}) é responsável por:
  \begin{itemize}
    \item configurar a segurança no nível do sistema,
    \item ancorar o secure boot (cadeia de confiança),
    \item estabelecer o \textbf{SPE} (\textit{Secure Processing Environment}).
  \end{itemize}
  \item Modelo mínimo: dois ambientes isolados:
  \begin{itemize}
    \item \textbf{SPE}: hospeda partições seguras (crypto, attestation, secure storage).
    \item \textbf{NSPE} (\textit{Non-Secure Processing Environment}): aplicação ``normal''.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{PSA na prática: Analyze + Architect}
\small
\textbf{Analyze} --- usar threat model + STRIDE para identificar ameaças ao dispositivo:
\begin{itemize}
  \item Quais são os \textbf{assets}? (chaves, firmware, dados de sensores, identidade)
  \item Quais são as \textbf{trust boundaries}? (SPE/NSPE, rede, interface de debug, OTA)
  \item Quais ameaças STRIDE se aplicam a cada fronteira?
\end{itemize}

\vspace{0.3em}
\textbf{Architect} --- mapear ameaças aos 10 security goals:
\begin{itemize}
  \item Firmware malicioso via OTA? $\rightarrow$ \textbf{Secure Update} + \textbf{Anti-Rollback} + \textbf{Secure Boot}.
  \item Extração de chaves? $\rightarrow$ \textbf{Secure Storage} + \textbf{Isolation} + \textbf{Crypto Services}.
  \item Spoofing de dispositivo? $\rightarrow$ \textbf{Unique Identity} + \textbf{Attestation}.
\end{itemize}
\end{frame}

\begin{frame}{PSA na prática: Implement + Certify}
\textbf{Implement} --- referência aberta:
\begin{itemize}
  \item \textbf{TF-M} (\textit{Trusted Firmware-M}): implementação open-source do PSA para Cortex-M.
  \item Fornece serviços prontos: crypto, attestation, secure storage, isolamento via partições.
  \item Portável para diferentes chips e plataformas.
\end{itemize}

\vspace{0.4em}
\textbf{Certify} --- níveis de certificação PSA Certified:
\begin{itemize}
  \item \textbf{Nível 1}: questionário de boas práticas (baseline).
  \item \textbf{Nível 2}: avaliação laboratorial do RoT contra ataques de software.
  \item \textbf{Nível 3}: resistência a ataques de hardware (side-channel, fault injection).
\end{itemize}
\end{frame}

\small
\textbf{Cenário:} sensor de temperatura industrial conectado via rede sem fio a um gateway, com atualização OTA e envio de dados para a nuvem.

\vspace{0.3em}
\begin{enumerate}

\begin{frame}{Resumo (com densidade)}
\begin{itemize}
  \item Use uma \textbf{taxonomia} para organizar atributos, ameaças e meios (dependability).
  \item Safety: de hazard analysis $\rightarrow$ constraints $\rightarrow$ requisitos $\rightarrow$ evidência.
  \item Security: threat modeling + risk assessment + controles + verificação contínua.
  \item Resiliência: projetar para \textbf{falhar com segurança} e \textbf{recuperar} (antecipar, resistir, recuperar, adaptar).
  \item Engenheiros justificam decisões via \textbf{trade-offs} e \textbf{evidência}.
\end{itemize}
\end{frame}
